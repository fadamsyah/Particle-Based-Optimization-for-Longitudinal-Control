def get_array_from_time_interval(x, t, T):
    t0 = min(T)
    t1 = max(T)
    i1 = -1
    i0 = 0
    for i in range(len(t)):
        if t0 <= t[i]:
            i0 = deepcopy(i)
            break
    for i in range(i0, len(t)):
        if t1 <= t[i]:
            i1 = deepcopy(i)
            break

    return x[i0:i1], t[i0:i1]
